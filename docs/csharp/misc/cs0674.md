 <span data-ttu-id="90e26-105">Il compilatore C# non consente l'uso di <xref:System.ParamArrayAttribute?displayProperty=nameWithType>; usare [params](../../csharp/language-reference/keywords/params.md) .</span><span class="sxs-lookup"><span data-stu-id="90e26-105">The C# compiler does not allow for the use of <xref:System.ParamArrayAttribute?displayProperty=nameWithType>; use [params](../../csharp/language-reference/keywords/params.md) instead.</span></span>  
  
 <span data-ttu-id="90e26-106">L'esempio seguente genera l'errore CS0674:</span><span class="sxs-lookup"><span data-stu-id="90e26-106">The following sample generates CS0674:</span></span>  
  
```csharp  
// CS0674.cs  
using System;  
public class MyClass   
{  
  
   public static void UseParams([ParamArray] int[] list)   // CS0674

   // public static void UseParams(params int[] list)   
   {  
      for ( int i = 0 ; i < list.Length ; i++ )  
         Console.WriteLine(list[i]);  
      Console.WriteLine();  
   }  
  
   public static void Main()   
   {  
      UseParams(1, 2, 3);  
   }  
}  
```
