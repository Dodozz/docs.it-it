 <span data-ttu-id="d8f54-106">L'esempio seguente genera l'errore CS0073:</span><span class="sxs-lookup"><span data-stu-id="d8f54-106">The following sample generates CS0073:</span></span>  
  
```csharp  
// CS0073.cs  
delegate void del();  
  
class Test  
{  
   public event del MyEvent  
   {  
      add;   // CS0073  
      // try the following lines instead  
      // add  
      // {  
      //    MyEvent += value;  
      // }  
      remove  
      {  
         MyEvent -= value;  
      }  
  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
