## <a name="example"></a><span data-ttu-id="2a6d2-110">Esempio</span><span class="sxs-lookup"><span data-stu-id="2a6d2-110">Example</span></span>

```csharp  
{  
   public static void Main()  
   {  
      int a = new int[5][,][][5;   // CS0178  
      int[,] b = new int[3,2];   // OK

      c[0] = new int[5][5];   // CS0178  
      c[0] = new int[2];   // OK  
      c[1] = new int[2]{1,2};   // OK  
   }  
}  
```
